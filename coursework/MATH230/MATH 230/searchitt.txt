char * searchit(char * string, int str_size, char ref_char)
{
    int j = 0;
    char * loc = &string[0];
    printf( "\n" );
    //For loop bi-conditional to pick first char found and break loop.
    for( j = 0; string[j] != '\0' && string[j] != ref_char ; j++ )
    {
      if(string[j] ==  ref_char)
      {
          loc = &string[j];
      }
    }
    if( *loc != ref_char )
      {
        loc = &string[j];
      }

}
